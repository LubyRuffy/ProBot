Dokku installion:
wget https://raw.githubusercontent.com/dokku/dokku/v0.7.2/bootstrap.sh
sudo DOKKU_TAG=v0.7.2 bash bootstrap.sh
go to the server's ip and finish the setup 

ON THE SERVER:
step1: create an app
dokku apps:create probot

step2: make a persistent storage an give the right permissions
sudo mkdir -p  /var/lib/dokku/data/probot-storage
sudo chown -R dokku:dokku /var/lib/dokku/data/probot-storage
sudo chown -R 32767:32767 /var/lib/dokku/data/probot-storage
sudo chmod a+rw /var/lib/dokku/data/probot-storage
dokku storage:mount probot /var/lib/dokku/data/probot-storage:/storage

step3: config the enviromment variables
dokku config:set probot APP_SETTINGS="config.DevelopmentConfig" DATABASE_URL="/storage/__database.db" CONTACT_EMAIL=probotmiti@gmail.com LOGGING_URL=localhost:514 APP_MAIL_USERNAME=probotmiti@gmail.com APP_MAIL_PASSWORD=probotindignado
(it's gonna fail but dont hurry...)

step4: create the database
dokku run probot python manage.py db init    --directory '/storage/migrations'
dokku run probot python manage.py db migrate --directory '/storage/migrations'
dokku run probot python manage.py db upgrade --directory '/storage/migrations'

ON THE CLIENT: push the apps files to the server
cat ~/.ssh/id_rsa.pub | ssh robot@robotcommander "sudo sshcommand acl-add dokku [description]"
git remote add dokku dokku@robotcommander:probot
git add .
git commit -a -m "first commit"
git push dokku master
